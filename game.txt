<script>
// Floating hearts
for(let i=0;i<60;i++){
  let heart=document.createElement('div');
  heart.classList.add('heart');
  heart.style.left=Math.random()*100+'%';
  heart.style.top=Math.random()*100+'%';
  heart.style.fontSize=(10+Math.random()*30)+'px';
  heart.style.animationDuration=(5+Math.random()*10)+'s';
  heart.style.opacity=Math.random()*0.4+0.2;
  heart.innerHTML='ðŸŽ€';
  document.body.appendChild(heart);
}

// Game logic
const thoughts = document.querySelectorAll('.thought');
const ribbon = document.getElementById('ribbon-bin');
const sheStayed = document.getElementById('she-stayed');
const scrollBtn = document.querySelector('.scroll-btn');
let collected = 0;

// Fixed scattered positions for all lines (avoids overlap)
const positions = [
  {left:'5vw', top:'50px'},
  {left:'25vw', top:'120px'},
  {left:'45vw', top:'70px'},
  {left:'65vw', top:'150px'},
  {left:'80vw', top:'100px'},
  {left:'10vw', top:'220px'},
  {left:'35vw', top:'200px'},
  {left:'55vw', top:'250px'},
  {left:'70vw', top:'300px'},
  {left:'20vw', top:'320px'},
  {left:'50vw', top:'400px'}
];

thoughts.forEach((t,i)=>{
  t.style.left = positions[i].left;
  t.style.top = positions[i].top;
});

// Drag functionality
thoughts.forEach(t=>{
  t.addEventListener('mousedown', e=>{
    e.preventDefault();
    let shiftX = e.clientX - t.getBoundingClientRect().left;
    let shiftY = e.clientY - t.getBoundingClientRect().top;
    t.style.cursor='grabbing';
    t.style.zIndex=1000;

    function moveAt(pageX,pageY){
      t.style.left = pageX - shiftX + 'px';
      t.style.top = pageY - shiftY + 'px';
    }

    function onMouseMove(event){
      moveAt(event.pageX,event.pageY);
    }

    document.addEventListener('mousemove',onMouseMove);

    document.addEventListener('mouseup', function upHandler(){
      document.removeEventListener('mousemove',onMouseMove);
      document.removeEventListener('mouseup',upHandler);
      t.style.cursor='grab';

      const ribbonRect = ribbon.getBoundingClientRect();
      const elemRect = t.getBoundingClientRect();
      if(
        elemRect.left + elemRect.width/2 > ribbonRect.left &&
        elemRect.right - elemRect.width/2 < ribbonRect.right &&
        elemRect.top + elemRect.height/2 > ribbonRect.top &&
        elemRect.bottom - elemRect.height/2 < ribbonRect.bottom
      ){
        t.style.display='none';
        collected++;
        if(collected>=thoughts.length){
          ribbon.style.display='none';
          sheStayed.style.opacity='1';
          sheStayed.style.transform='translateX(-50%) scale(1)';
          scrollBtn.style.opacity='1';
          scrollBtn.style.transform='scale(1)';
        }
      }
    });
  });
});

document.ondragstart = ()=>false;
